# Fressh mdev Bridge Mobile Integration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Route generic Workmux mobile commands through the persistent typed `mdev bridge --jsonl` protocol, fail noisily when the bridge is unavailable, and keep DirectMux as the scroll path.

**Architecture:** The Workmux control-channel layer remains the transport boundary. UI and keyboard code continue to express intent with existing argv builders, a mapper converts supported argv shapes to typed bridge operations, and a lazy per-connection bridge client owns JSONL startup, capability checks, request sequencing, response parsing, and disposal. The command stream API gets the smallest native/TypeScript addition needed to write JSONL requests to remote stdin.

**Tech Stack:** Expo React Native, TypeScript, `@fressh/react-native-uniffi-russh`, Rust UniFFI, russh, node:test integration tests, Jest wrapper tests.

impl CommandStreamSession {
    pub fn get_info(&self) -> CommandStreamInfo {
        self.info.clone()
    }

    pub async fn send_data(&self, data: Vec<u8>) -> Result<(), SshError> {
        let w = self.writer.lock().await;
        w.data(&data[..]).await?;
        Ok(())
    }

    pub async fn close(&self) -> Result<(), SshError> {
        self.close_internal().await
    }
}

```ts
export type MdevBridgeOperationRequest = {
	operation: string;
	params: Record<string, string | number>;
};

export const WORKMUX_REQUIRED_MDEV_BRIDGE_OPERATIONS = [
	'tmux.app.context',
	'tmux.app.window',
	'tmux.app.focus',
	'tmux.app.nav',
	'tmux.app.notification.open',
	'tmux.nav',
] as const;

const WORKMUX_NAV_ACTIONS = new Set(['next', 'prev', 'next-all', 'prev-all']);

function unsupported(argv: string[]): never {
	throw new Error(`Unsupported Workmux bridge command: ${JSON.stringify(argv)}`);
}

function isSafeNonNegativeIntegerText(value: string): boolean {
	if (!/^(0|[1-9]\d*)$/.test(value)) {
		return false;
	}
	return Number.isSafeInteger(Number(value));
}

function parseSelectIndex(value: string, argv: string[]): number {
	if (!isSafeNonNegativeIntegerText(value)) {
		unsupported(argv);
	}
	return Number(value);
}

export function buildMdevBridgeOperationFromWorkmuxArgv(
	argv: string[],
): MdevBridgeOperationRequest {
	const [scope, area, command] = argv;

	if (scope === 'tmux' && area === 'app') {
		if (
			command === 'context' &&
			argv.length === 5 &&
			argv[3] === '--session'
		) {
			return {
				operation: 'tmux.app.context',
				params: { session: argv[4] },
			};
		}

		if (
			command === 'window' &&
			argv.length === 5 &&
			argv[3] === '--session'
		) {
			return {
				operation: 'tmux.app.window',
				params: { session: argv[4] },
			};
		}

		if (
			command === 'notification' &&
			argv.length === 8 &&
			argv[3] === 'open' &&
			argv[4] === '--session' &&
			argv[6] === '--window-id'
		) {
			return {
				operation: 'tmux.app.notification.open',
				params: { session: argv[5], windowId: argv[7] },
			};
		}

		if (
			command === 'focus' &&
			argv.length === 6 &&
			argv[4] === '--session'
		) {
			return {
				operation: 'tmux.app.focus',
				params: { roleOrDirection: argv[3], session: argv[5] },
			};
		}

		if (command === 'nav' && argv.length === 6 && argv[4] === '--session') {
			if (!WORKMUX_NAV_ACTIONS.has(argv[3])) {
				unsupported(argv);
			}
			return {
				operation: 'tmux.app.nav',
				params: { action: argv[3], session: argv[5] },
			};
		}

		if (
			command === 'nav' &&
			argv.length === 7 &&
			argv[3] === 'select' &&
			argv[5] === '--session'
		) {
			return {
				operation: 'tmux.app.nav',
				params: {
					action: 'select',
					index: parseSelectIndex(argv[4], argv),
					session: argv[6],
				},
			};
		}
	}

	if (
		scope === 'tmux' &&
		area === 'nav' &&
		command === 'cycle' &&
		argv.length === 4
	) {
		return {
			operation: 'tmux.nav',
			params: { action: 'cycle', target: argv[3] },
		};
	}

	unsupported(argv);
}
```

```ts
import { type WorkmuxControlCommandResult } from './workmux-control-channel';

export type MdevBridgeStreamEvent =
	| { type: 'stdout'; bytes: ArrayBuffer }
	| { type: 'stderr'; bytes: ArrayBuffer }
	| { type: 'exitStatus'; exitStatus: number }
	| { type: 'exitSignal'; signalName: string }
	| { type: 'closed' };

export type MdevBridgeCommandStream = {
	sendData: (data: ArrayBuffer, opts?: { signal?: AbortSignal }) => Promise<void>;
	close: (opts?: { signal?: AbortSignal }) => Promise<void>;
};

export type MdevBridgeStreamConnection = {
	startCommandStream: (opts: {
		command: string;
		onEvent: (event: MdevBridgeStreamEvent) => void;
		abortSignal?: AbortSignal;
	}) => Promise<MdevBridgeCommandStream>;
};

export type MdevBridgeClient = {
	runOperation: (input: {
		operation: string;
		params: Record<string, unknown>;
		timeoutMs?: number;
	}) => Promise<WorkmuxControlCommandResult>;
	dispose: () => Promise<void>;
};

export const MDEV_BRIDGE_UPDATE_MESSAGE =
	'Update mdev on the remote machine; this action requires mdev bridge --jsonl.';

const MDEV_BRIDGE_COMMAND = 'mdev bridge --jsonl';
const MDEV_BRIDGE_PROTOCOL_ERROR = 'mdev bridge protocol error.';
const MDEV_BRIDGE_STREAM_CLOSED = 'mdev bridge stream closed.';
const MDEV_BRIDGE_TIMEOUT = 'mdev bridge request timed out.';
const DEFAULT_MDEV_BRIDGE_REQUEST_TIMEOUT_MS = 10_000;

type PendingRequest = {
	id: string;
	timer: ReturnType<typeof setTimeout>;
	resolve: (result: WorkmuxControlCommandResult) => void;
};

function encodeLine(value: unknown): ArrayBuffer {
	return new TextEncoder().encode(`${JSON.stringify(value)}\n`).buffer;
}

function decodeBytes(bytes: ArrayBuffer): string {
	return new TextDecoder().decode(bytes);
}

function isRecord(value: unknown): value is Record<string, unknown> {
	return typeof value === 'object' && value !== null && !Array.isArray(value);
}

function successOutput(value: unknown): string {
	return `${JSON.stringify(value ?? {})}\n`;
}

function failure(error: string): WorkmuxControlCommandResult {
	return { success: false, output: '', error };
}

function success(result: unknown): WorkmuxControlCommandResult {
	return { success: true, output: successOutput(result) };
}

function validateHello(
	value: unknown,
	requiredOperations: readonly string[],
): boolean {
	if (!isRecord(value)) return false;
	if (value.ok !== true) return false;
	if (value.protocolVersion !== 1) return false;
	if (!Array.isArray(value.supportedRequestTypes)) return false;
	if (!value.supportedRequestTypes.includes('operation')) return false;
	if (!Array.isArray(value.operations)) return false;
	return requiredOperations.every((operation) =>
		value.operations.includes(operation),
	);
}

export function createMdevBridgeClient({
	connection,
	requiredOperations,
	requestTimeoutMs = DEFAULT_MDEV_BRIDGE_REQUEST_TIMEOUT_MS,
}: {
	connection: MdevBridgeStreamConnection | null;
	requiredOperations: readonly string[];
	requestTimeoutMs?: number;
}): MdevBridgeClient {
	let stream: MdevBridgeCommandStream | null = null;
	let ready = false;
	let disposed = false;
	let failedError: string | null = null;
	let stdoutBuffer = '';
	let nextRequestId = 1;
	let pending: PendingRequest | null = null;
	let queue = Promise.resolve();

	const failPending = (error: string): void => {
		failedError = error;
		if (!pending) return;
		clearTimeout(pending.timer);
		const current = pending;
		pending = null;
		current.resolve(failure(error));
	};

	const handleLine = (line: string): void => {
		let parsed: unknown;
		try {
			parsed = JSON.parse(line);
		} catch {
			failPending(MDEV_BRIDGE_PROTOCOL_ERROR);
			return;
		}

		if (!pending || !isRecord(parsed) || parsed.id !== pending.id) {
			failPending(MDEV_BRIDGE_PROTOCOL_ERROR);
			return;
		}

		clearTimeout(pending.timer);
		const current = pending;
		pending = null;

		if (current.id === 'mdev-bridge-1') {
			if (!validateHello(parsed, requiredOperations)) {
				current.resolve(failure(MDEV_BRIDGE_UPDATE_MESSAGE));
				failedError = MDEV_BRIDGE_UPDATE_MESSAGE;
				return;
			}
			ready = true;
			current.resolve(success({}));
			return;
		}

		if (parsed.ok === true) {
			current.resolve(success(parsed.result));
			return;
		}

		current.resolve(
			failure(typeof parsed.error === 'string' ? parsed.error : 'mdev bridge operation failed.'),
		);
	};

	const onEvent = (event: MdevBridgeStreamEvent): void => {
		if (event.type === 'stdout') {
			stdoutBuffer += decodeBytes(event.bytes);
			for (;;) {
				const newlineIndex = stdoutBuffer.indexOf('\n');
				if (newlineIndex < 0) break;
				const line = stdoutBuffer.slice(0, newlineIndex).trim();
				stdoutBuffer = stdoutBuffer.slice(newlineIndex + 1);
				if (line.length > 0) handleLine(line);
			}
			return;
		}
		if (event.type === 'closed' || event.type === 'exitStatus' || event.type === 'exitSignal') {
			failPending(MDEV_BRIDGE_STREAM_CLOSED);
		}
	};

	const sendRequest = async (
		request: Record<string, unknown>,
		timeoutMs: number,
	): Promise<WorkmuxControlCommandResult> => {
		if (!stream || failedError) {
			return failure(failedError ?? MDEV_BRIDGE_UPDATE_MESSAGE);
		}
		return new Promise((resolve) => {
			const id = String(request.id);
			const timer = setTimeout(() => {
				if (pending?.id !== id) return;
				pending = null;
				failedError = MDEV_BRIDGE_TIMEOUT;
				resolve(failure(MDEV_BRIDGE_TIMEOUT));
			}, timeoutMs);
			pending = { id, timer, resolve };
			stream
				.sendData(encodeLine(request))
				.catch(() => failPending(MDEV_BRIDGE_STREAM_CLOSED));
		});
	};

	const ensureReady = async (): Promise<WorkmuxControlCommandResult> => {
		if (ready) return success({});
		if (failedError) return failure(failedError);
		if (!connection) return failure(MDEV_BRIDGE_UPDATE_MESSAGE);
		if (!stream) {
			try {
				stream = await connection.startCommandStream({
					command: MDEV_BRIDGE_COMMAND,
					onEvent,
				});
			} catch {
				failedError = MDEV_BRIDGE_UPDATE_MESSAGE;
				return failure(MDEV_BRIDGE_UPDATE_MESSAGE);
			}
		}
		return sendRequest({ id: 'mdev-bridge-1', type: 'hello' }, requestTimeoutMs);
	};

	const runQueued = async (input: {
		operation: string;
		params: Record<string, unknown>;
		timeoutMs?: number;
	}): Promise<WorkmuxControlCommandResult> => {
		if (disposed) return failure('mdev bridge client disposed.');
		const startup = await ensureReady();
		if (!startup.success) return startup;
		const id = `mdev-bridge-${++nextRequestId}`;
		return sendRequest(
			{
				id,
				type: 'operation',
				operation: input.operation,
				params: input.params,
				timeoutMs: input.timeoutMs ?? requestTimeoutMs,
			},
			input.timeoutMs ?? requestTimeoutMs,
		);
	};

	return {
		runOperation: (input) => {
			const next = queue.then(() => runQueued(input));
			queue = next.then(
				() => undefined,
				() => undefined,
			);
			return next;
		},
		dispose: async () => {
			disposed = true;
			failPending('mdev bridge client disposed.');
			await stream?.close();
		},
	};
}
```

			? successResult()

			exit: (input) =>

		},
		dispose: async () => {
			disposed = true;
			await bridgeClient.dispose();
			await directTmuxTransport.dispose();
